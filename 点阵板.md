void showHanzi(uint8_t* hz)
{
	uint8_t logoCmd[6]=
	{
		0x25,//长度
		0x12,//设备号，在模块复位时上报。12C18ECB
		0xC1,
		0x8E,
		0xCB,
		0x07,//命令号
	//点阵数据
	};
	sendCmd(logoCmd,6);
	sendCmd(hz,32);
}
void main()
{
	uint8_t i;
	while(1)
	{
		for(i=0;i＜32;i++)
		{
			showHanzi((uint8_t*)zqkl+((32-i)*4));
			Delay100ms();Delay100ms();
		}
	};
}
void Delay104us()
//@33.1776MHz
{
	unsigned char i, j;
	_nop_();
	_nop_();
	i = 4;
	j = 87;
	do
	{
	while (--j);
	} while (--i);
}
void sendByte(uint8_t byte)
{
	uint8_t i;
	SWTXD = 0;
	Delay104us();
	for(i=0;i＜8;i++)
	{
		byte ＞＞= 1;
		SWTXD = CY;
		Delay104us();
	}
	SWTXD = 1;
	Delay104us();
}
void sendCmd(uint8_t* cmd, uint8_t len)
{
	while(len--)
	{
		sendByte(*cmd);
		cmd++;
	}
}

		while (--k);
		} while (--j);
	} while (--i);
}
